166 void T2DPhysicCollider::render(sf::RenderWindow& window, class T2DPhysicBody body)
167 {
168  if (type == collider_types::CIRCULAR)
169  {
170 
171  sf::CircleShape circle;
172  circle.setRadius(shape.circular_shape.m_radius);
173 
174  b2Vec2 center = b2Mul(body.get_body()->GetTransform(), shape.circular_shape.m_p);
175 
176  circle.setPosition({ center.x - circle.getRadius(), window.getSize().y - center.y - circle.getRadius() });
177  circle.setFillColor(render_color);
178 
179  window.draw(circle);
180  }
181 
182  else if (type == collider_types::POLYGON)
183  {
184 
185  sf::ConvexShape convex;
186  convex.setPointCount(4);
187 
188  b2Vec2 first = b2Mul(body.get_body()->GetTransform(), shape.polygon_shape.GetVertex(0));
189  b2Vec2 second = b2Mul(body.get_body()->GetTransform(), shape.polygon_shape.GetVertex(1));
190  b2Vec2 third = b2Mul(body.get_body()->GetTransform(), shape.polygon_shape.GetVertex(2));
191  b2Vec2 fourth = b2Mul(body.get_body()->GetTransform(), shape.polygon_shape.GetVertex(3));
192 
193  convex.setPoint(0, { first.x, window.getSize().y - first.y });
194  convex.setPoint(1, { second.x, window.getSize().y - second.y });
195  convex.setPoint(2, { third.x, window.getSize().y - third.y });
196  convex.setPoint(3, { fourth.x, window.getSize().y - fourth.y });
197 
198  convex.setFillColor(render_color);
199  window.draw(convex);
200  }
201 
202  else if (type == collider_types::EDGE)
203  {
204 
205  b2Vec2 origin = b2Mul(body.get_body()->GetTransform(), shape.edge_shape.m_vertex1);
206  b2Vec2 end = b2Mul(body.get_body()->GetTransform(), shape.edge_shape.m_vertex2);
207 
208  sf::Vertex edge[] =
209  {
210  sf::Vertex(sf::Vector2f{ origin.x, window.getSize().y - origin.y}, render_color),
211  sf::Vertex(sf::Vector2f{ end.x, window.getSize().y - end.y}, render_color)
212  };
213 
214  window.draw(edge, 2, sf::Lines);
215  }
216 }
